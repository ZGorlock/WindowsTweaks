Launch: Task Manager
	Switch to the Details Tab
	Right Click 'TextInputHost.exe'
	Select End Task


Navigate to: 'C:\Windows\SystemApps\MicrosoftWindows.Client.CBS_cw5n1h2txyewy'

	Rename 'C:\Windows\SystemApps\MicrosoftWindows.Client.CBS_cw5n1h2txyewy\InputApp\'
		to 'C:\Windows\SystemApps\MicrosoftWindows.Client.CBS_cw5n1h2txyewy\InputApp.bak\'

	Rename 'C:\Windows\SystemApps\MicrosoftWindows.Client.CBS_cw5n1h2txyewy\TextInputHost.exe'
		to 'C:\Windows\SystemApps\MicrosoftWindows.Client.CBS_cw5n1h2txyewy\TextInputHost.exe.bak'
